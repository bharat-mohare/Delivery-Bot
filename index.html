#include #include #include #include #include #include // ===================================================== // WIFI MANAGER // ===================================================== WiFiManager wm; // ===================================================== // WEBSOCKET // ===================================================== WebSocketsServer webSocket(81); // ===================================================== // MOTOR PINS // ===================================================== // RIGHT SIDE #define IN1 18 #define IN2 19 // LEFT SIDE #define IN3 25 #define IN4 33 // ===================================================== // ULTRASONIC // ===================================================== #define TRIG_F 4 #define ECHO_F 16 // ===================================================== // ENCODER // ===================================================== #define ENCODER_PIN 26 volatile long pulseCount = 0; volatile unsigned long lastPulse = 0; // ===================================================== // WHEEL // ===================================================== float wheelCircumference = 35.5; // 8 black marker lines int slots = 8; float cmPerPulse = wheelCircumference / slots; // ===================================================== // MPU6050 // ===================================================== MPU6050 mpu6050(Wire); // ===================================================== // CONTROL // ===================================================== String command = "X"; unsigned long lastCommandTime = 0; // ===================================================== // ENCODER INTERRUPT // ===================================================== void IRAM_ATTR countPulse() { unsigned long now = micros(); // strong debounce if (now - lastPulse > 50000) { pulseCount++; lastPulse = now; } } // ===================================================== // MOTOR FUNCTIONS // ===================================================== void stopBot() { digitalWrite(IN1, LOW); digitalWrite(IN2, LOW); digitalWrite(IN3, LOW); digitalWrite(IN4, LOW); } void forwardBot() { digitalWrite(IN1, HIGH); digitalWrite(IN2, LOW); digitalWrite(IN3, HIGH); digitalWrite(IN4, LOW); } void backwardBot() { digitalWrite(IN1, LOW); digitalWrite(IN2, HIGH); digitalWrite(IN3, LOW); digitalWrite(IN4, HIGH); } void leftBot() { digitalWrite(IN1, HIGH); digitalWrite(IN2, LOW); digitalWrite(IN3, LOW); digitalWrite(IN4, HIGH); } void rightBot() { digitalWrite(IN1, LOW); digitalWrite(IN2, HIGH); digitalWrite(IN3, HIGH); digitalWrite(IN4, LOW); } // ===================================================== // DISTANCE // ===================================================== float getDistanceCM() { return pulseCount * cmPerPulse; } // ===================================================== // ULTRASONIC // ===================================================== long getFrontDistance() { digitalWrite(TRIG_F, LOW); delayMicroseconds(2); digitalWrite(TRIG_F, HIGH); delayMicroseconds(10); digitalWrite(TRIG_F, LOW); long duration = pulseIn(ECHO_F, HIGH, 30000); return duration * 0.034 / 2; } // ===================================================== // FORWARD DISTANCE // ===================================================== void moveForwardCM(float targetCM) { pulseCount = 0; stopBot(); delay(300); while (getDistanceCM() < targetCM) { long front = getFrontDistance(); if (front > 0 && front < 20) { stopBot(); Serial.println("OBSTACLE"); return; } forwardBot(); mpu6050.update(); Serial.print("DISTANCE: "); Serial.println(getDistanceCM()); delay(5); } stopBot(); delay(300); } // ===================================================== // BACKWARD DISTANCE // ===================================================== void moveBackwardCM(float targetCM) { pulseCount = 0; stopBot(); delay(300); while (getDistanceCM() < targetCM) { backwardBot(); mpu6050.update(); Serial.print("DISTANCE: "); Serial.println(getDistanceCM()); delay(5); } stopBot(); delay(300); } // ===================================================== // LEFT TURN // ===================================================== void turnLeftDeg(float targetAngle) { stopBot(); delay(500); mpu6050.update(); float startAngle = mpu6050.getAngleZ(); float currentAngle = 0; while (currentAngle < targetAngle) { mpu6050.update(); currentAngle = abs( mpu6050.getAngleZ() - startAngle ); leftBot(); Serial.print("LEFT: "); Serial.println(currentAngle); delay(8); } stopBot(); delay(500); } // ===================================================== // RIGHT TURN // ===================================================== void turnRightDeg(float targetAngle) { stopBot(); delay(500); mpu6050.update(); float startAngle = mpu6050.getAngleZ(); float currentAngle = 0; while (currentAngle < targetAngle) { mpu6050.update(); currentAngle = abs( mpu6050.getAngleZ() - startAngle ); rightBot(); Serial.print("RIGHT: "); Serial.println(currentAngle); delay(8); } stopBot(); delay(500); } // ===================================================== // ROUTE EXECUTION // ===================================================== void executeRoute(String route) { int startIndex = 0; while (startIndex < route.length()) { int commaIndex = route.indexOf(',', startIndex); if (commaIndex == -1) { commaIndex = route.length(); } String cmd = route.substring( startIndex, commaIndex); cmd.trim(); char action = cmd.charAt(0); float value = cmd.substring(1).toFloat(); if (action == 'F') { moveForwardCM(value); } else if (action == 'B') { moveBackwardCM(value); } else if (action == 'L') { turnLeftDeg(value); } else if (action == 'R') { turnRightDeg(value); } delay(500); startIndex = commaIndex + 1; } stopBot(); } // ===================================================== // WEBSOCKET EVENT // ===================================================== void webSocketEvent( uint8_t num, WStype_t type, uint8_t * payload, size_t length) { if (type == WStype_TEXT) { String msg = (char*)payload; Serial.println(msg); // RESET WIFI if (msg == "RESET_WIFI") { Serial.println( "RESETTING WIFI"); wm.resetSettings(); delay(1000); ESP.restart(); return; } // AUTOMATION if (msg.indexOf("F") >= 0 || msg.indexOf("L") >= 0 || msg.indexOf("R") >= 0 || msg.indexOf("B") >= 0) { executeRoute(msg); return; } command = msg; lastCommandTime = millis(); } } // ===================================================== // SETUP // ===================================================== void setup() { Serial.begin(115200); // ================================================= // SAFE MOTOR STARTUP // ================================================= pinMode(IN1, OUTPUT); digitalWrite(IN1, LOW); pinMode(IN2, OUTPUT); digitalWrite(IN2, LOW); pinMode(IN3, OUTPUT); digitalWrite(IN3, LOW); pinMode(IN4, OUTPUT); digitalWrite(IN4, LOW); stopBot(); delay(3000); // ================================================= // ULTRASONIC // ================================================= pinMode(TRIG_F, OUTPUT); pinMode(ECHO_F, INPUT); // ================================================= // ENCODER // ================================================= pinMode( ENCODER_PIN, INPUT_PULLUP); attachInterrupt( digitalPinToInterrupt( ENCODER_PIN), countPulse, FALLING); // ================================================= // MPU6050 // ================================================= Wire.begin(21, 22); delay(1000); mpu6050.begin(); Serial.println( "KEEP ROBOT STILL"); mpu6050.calcGyroOffsets(true); Serial.println( "GYRO READY"); // ================================================= // AP + STA MODE // ================================================= WiFi.mode(WIFI_AP_STA); // permanent setup hotspot WiFi.softAP( "DeliveryBot_Setup"); Serial.println( "SETUP AP ACTIVE"); Serial.println( WiFi.softAPIP()); // ================================================= // WIFI MANAGER // ================================================= bool result = wm.autoConnect( "DeliveryBot_Setup"); if (!result) { Serial.println( "WIFI FAILED"); } else { Serial.println( "WIFI CONNECTED"); Serial.println( WiFi.localIP()); } // ================================================= // MDNS // ================================================= if (MDNS.begin("robot")) { Serial.println( "MDNS STARTED"); Serial.println( "robot.local"); } // ================================================= // WEBSOCKET // ================================================= webSocket.begin(); webSocket.onEvent( webSocketEvent); Serial.println( "ROBOT READY"); } // ===================================================== // LOOP // ===================================================== void loop() { webSocket.loop(); mpu6050.update(); // auto stop if (millis() - lastCommandTime > 200) { command = "X"; } if (command == "W") { forwardBot(); } else if (command == "S") { backwardBot(); } else if (command == "A") { leftBot(); } else if (command == "D") { rightBot(); } else { stopBot(); } delay(5); }
